What are Decorators?

Stage 3

Decorators are a proposal for extending JavaScript classes. They allow you to annotate and modify classes and their members at definition time. The Stage 3 proposal (TC39) introduces a standardized decorator syntax that differs significantly from the legacy (experimental) TypeScript/Babel decorators.

Key Concepts

Class Decorators

Class decorators receive the class constructor and a context object with kind: "class". They can return a new class (subclass) or modify the class via addInitializer.

// Sealed class decorator function sealed(value, { kind, name }) { if (kind !== "class") throw new TypeError("@sealed for classes only"); Object.seal(value); Object.seal(value.prototype); console.log(`Class ${name} is now sealed`); return value; } // Singleton decorator — wraps class with a proxy function singleton(value, { kind, name }) { if (kind !== "class") throw new TypeError("@singleton for classes only"); let instance = null; return class extends value { constructor(...args) { if (instance) { console.log(`Returning existing ${name} instance`); return instance; } super(...args); instance = this; console.log(`Created new ${name} instance`); } }; } @singleton class Database { constructor(url) { this.url = url; } } const db1 = new Database("postgres://localhost"); const db2 = new Database("postgres://other"); console.log("Same instance?", db1 === db2); console.log("URL:", db1.url);

Class with addInitializer

The addInitializer callback runs after the class is fully defined. Useful for registration patterns.

const registry = new Map(); function register(value, { kind, name, addInitializer }) { if (kind !== "class") return; addInitializer(function() { // 'this' is the class itself registry.set(name, this); console.log(`Registered: ${name}`); }); } @register class UserService {} @register class AuthService {} console.log("Registry entries:", [...registry.keys()]); console.log("UserService found:", registry.get("UserService") === UserService);

Method Decorators

Method decorators receive the original method function and can return a replacement. The context has kind: "method" and provides the method name, static flag, and access object.

// Logging decorator — wraps a method to log calls function log(value, { kind, name }) { if (kind !== "method") return; return function(...args) { console.log(`>> ${name}(${args.join(", ")})`); const result = value.call(this, ...args); console.log(`<< ${name} => ${JSON.stringify(result)}`); return result; }; } // Bound decorator — auto-binds method to instance function bound(value, { kind, name, addInitializer }) { if (kind !== "method") return; addInitializer(function() { this[name] = this[name].bind(this); }); } class Calculator { result = 0; @log add(a, b) { return a + b; } @log @bound multiply(a, b) { this.result = a * b; return this.result; } } const calc = new Calculator(); calc.add(2, 3); // Test bound — destructured still works const { multiply } = calc; multiply(4, 5); console.log("calc.result:", calc.result);

Timing & Retry Decorators

// Timed decorator — measures execution time function timed(value, { kind, name }) { if (kind !== "method") return; return function(...args) { const start = performance.now(); const result = value.call(this, ...args); const ms = (performance.now() - start).toFixed(3); console.log(`${name} took ${ms}ms`); return result; }; } // Deprecation warning decorator function deprecated(message) { return function(value, { kind, name }) { if (kind !== "method") return; let warned = false; return function(...args) { if (!warned) { console.warn(`DEPRECATED: ${name} — ${message}`); warned = true; } return value.call(this, ...args); }; }; } class DataProcessor { @timed heavyCompute(n) { let sum = 0; for (let i = 0; i < n; i++) sum += Math.sqrt(i); return sum; } @deprecated("Use processV2() instead") processV1(data) { return data.toUpperCase(); } } const proc = new DataProcessor(); proc.heavyCompute(100000); proc.processV1("hello"); proc.processV1("again"); // no second warning

Field Decorators

Field decorators receive undefined as their first argument (since the field value doesn't exist yet at decoration time). They can return an initializer function that transforms the initial value.

// Default value decorator function defaultValue(def) { return function(value, { kind, name }) { if (kind !== "field") return; return function(initialValue) { const v = initialValue ?? def; console.log(`Field ${name} initialized to: ${JSON.stringify(v)}`); return v; }; }; } // Type coercion decorator function coerce(type) { return function(value, { kind, name }) { if (kind !== "field") return; return function(initialValue) { const coerced = type(initialValue); console.log(`Field ${name}: ${typeof initialValue} → ${typeof coerced}`); return coerced; }; }; } class Config { @defaultValue("localhost") host; @defaultValue(3000) port; @coerce(Number) timeout = "5000"; @coerce(String) id = 42; } const cfg = new Config(); console.log(`host: ${cfg.host}, port: ${cfg.port}`); console.log(`timeout: ${cfg.timeout} (${typeof cfg.timeout})`); console.log(`id: ${cfg.id} (${typeof cfg.id})`);

Validation Field Decorators

// Range validator for field initialization function range(min, max) { return function(value, { kind, name }) { if (kind !== "field") return; return function(initialValue) { if (initialValue < min || initialValue > max) { console.error( `${name}: ${initialValue} out of range [${min}, ${max}]` ); return Math.max(min, Math.min(max, initialValue)); } console.log(`${name}: ${initialValue} is within range`); return initialValue; }; }; } // Freeze initial value function frozen(value, { kind, name }) { if (kind !== "field") return; return function(initialValue) { const result = Object.freeze(initialValue); console.log(`${name}: frozen`); return result; }; } class Player { @range(1, 100) health = 80; @range(0, 10) level = 15; // will be clamped @frozen stats = { str: 10, dex: 8, int: 12 }; } const player = new Player(); console.log(`Health: ${player.health}, Level: ${player.level}`); try { player.stats.str = 99; } catch(e) { console.error(e.message); }

Getter & Setter Decorators

Getter and setter decorators work similarly to method decorators. The context kind is "getter" or "setter".

// Lazy getter — compute once, cache forever function lazy(getter, { kind, name }) { if (kind !== "getter") return; const cache = new WeakMap(); return function() { if (!cache.has(this)) { console.log(`Computing ${name}...`); cache.set(this, getter.call(this)); } else { console.log(`Cached ${name}`); } return cache.get(this); }; } // Setter validation function validate(fn) { return function(setter, { kind, name }) { if (kind !== "setter") return; return function(value) { if (!fn(value)) { console.error(`Validation failed for ${name}: ${value}`); return; } setter.call(this, value); }; }; } class User { #firstName; #lastName; #age; constructor(first, last, age) { this.#firstName = first; this.#lastName = last; this.#age = age; } @lazy get fullName() { return `${this.#firstName} ${this.#lastName}`; } @validate(v => typeof v === "number" && v >= 0 && v <= 150) set age(v) { this.#age = v; } get age() { return this.#age; } } const user = new User("John", "Doe", 30); console.log(user.fullName); // computes console.log(user.fullName); // cached user.age = 25; console.log(`Age: ${user.age}`); user.age = -5; // validation fails console.log(`Age after bad set: ${user.age}`);

Auto-accessor Keyword

The accessor keyword creates a getter/setter pair backed by a private slot. Decorators for auto-accessors receive {get, set} and can return {get?, set?, init?}.

// Observable auto-accessor function observable({ get, set }, { kind, name }) { if (kind !== "accessor") return; return { get() { const value = get.call(this); console.log(`[GET] ${name} = ${JSON.stringify(value)}`); return value; }, set(value) { const old = get.call(this); set.call(this, value); console.log(`[SET] ${name}: ${JSON.stringify(old)} → ${JSON.stringify(value)}`); }, init(value) { console.log(`[INIT] ${name} = ${JSON.stringify(value)}`); return value; } }; } // Clamped auto-accessor function clamped(min, max) { return function({ get, set }, { kind, name }) { if (kind !== "accessor") return; return { set(value) { const v = Math.max(min, Math.min(max, value)); if (v !== value) console.warn(`${name} clamped: ${value} → ${v}`); set.call(this, v); }, init(value) { return Math.max(min, Math.min(max, value)); } }; }; } class Slider { @observable accessor label = "Volume"; @clamped(0, 100) accessor value = 50; } const s = new Slider(); s.label; // get logged s.label = "Bass"; // set logged s.value = 120; // clamped to 100 console.log(`value: ${s.value}`);

Decorator Metadata API

The metadata API allows decorators to attach data to context.metadata. After class definition, metadata is accessible via Symbol.metadata on the class. This enables frameworks to collect information about decorated members.

// Serialize decorator — marks fields for serialization function serialize(alias) { return function(value, { kind, name, metadata }) { if (!metadata.serializable) metadata.serializable = []; metadata.serializable.push({ field: name, alias: alias || name }); }; } // Endpoint decorator for methods function endpoint(method, path) { return function(value, { kind, name, metadata }) { if (!metadata.routes) metadata.routes = []; metadata.routes.push({ handler: name, method: method, path: path }); }; } class UserController { @serialize("user_name") name = "default"; @serialize() email = ""; @endpoint("GET", "/users") list() { return []; } @endpoint("POST", "/users") create(data) { return data; } } const meta = UserController[Symbol.metadata]; console.log("Serializable fields:", JSON.stringify(meta?.serializable, null, 2)); console.log("Routes:", JSON.stringify(meta?.routes, null, 2));

// Track decorator — stores method names in metadata function track(value, { kind, name, metadata }) { if (!metadata.tracked) metadata.tracked = []; metadata.tracked.push(name); } class Base { @track baseMethod() {} } class Child extends Base { @track childMethod() {} } const baseMeta = Base[Symbol.metadata]; const childMeta = Child[Symbol.metadata]; console.log("Base tracked:", baseMeta?.tracked); console.log("Child tracked:", childMeta?.tracked); console.log("Child inherits Base metadata:", Object.getPrototypeOf(childMeta) === baseMeta);
